pub fn get_bus_routes(client: &LTAClient) -> Result<Vec<BusRoute>>

Returns detailed route information for all services currently in operation, including: all bus stops along each route, first/last bus timings for each stop

Update freq: Ad-Hoc

Example

use lta::prelude::*;
use lta::bus::get_bus_routes;
use lta::lta_client::LTAClient;

fn main() -> Result<()> {
    let api_key = std::env::var("API_KEY").unwrap();
    let client = LTAClient::with_api_key(api_key);
    let bus_routes: Vec<BusRoute> = get_bus_routes(&client)?;
    println!("{:?}", bus_routes);
    Ok(())
}
